\section[\texthebrew{דו-כיוניות: הגדרה ומשמעות}]%
        {\texthebrew{דו-כיוניות: הגדרה ומשמעות}\mbox{\setLTR\enspace\thesection}}
\label{sec:bidi-meaning}

\begin{hebrew}
דו-כיוניות
(\foreignlanguage{english}{Bidirectionality},
בקיצור~\foreignlanguage{english}{BiDi})
היא יכולת של מערכת עיבוד טקסט לשלב בו-זמנית
שפות מימין-לשמאל
(\foreignlanguage{english}{RTL})
כגון עברית,
לצד שפות שמאל-לימין
(\foreignlanguage{english}{LTR})
כגון אנגלית,
בתוך אותה שורה או פסקה~\mbox{\cite{Unicode2023}}.
תקן \foreignlanguage{english}{Unicode}
מגדיר את האלגוריתם הדו-כיווני
(\foreignlanguage{english}{Unicode Bidirectional Algorithm},
ראשי-תיבות~\foreignlanguage{english}{UBA}),
הקובע את סדר הצגת התווים בשורה המעורבת.
לכל תו מוקצה ערך כיוון:
\foreignlanguage{english}{strong},
\foreignlanguage{english}{weak}
או~\foreignlanguage{english}{neutral};
לאחר הקצאה זו האלגוריתם מסדר מחדש את תצוגת התווים
בהתאם לכיוון הסביבה הפעיל~\mbox{\cite{Unicode2023}}.

במסגרת מערכות סוכנים מבוססות-\foreignlanguage{english}{LLM},
הדו-כיוניות מציבה שתי בעיות מרכזיות.
ראשית, \textbf{היפוך מונחים טכניים}:
כאשר סוכן מייצר טקסט עברי הכולל שמות פקודות
\foreignlanguage{english}{LaTeX},
מנוע ה-\foreignlanguage{english}{UBA}
עלול להציג את תווי הפקודה בסדר הפוך
אם לא נוסף סימון כיוון מפורש.
שנית, \textbf{היפוך מספרי ציטוטים}:
מספר רב-ספרתי כגון
\foreignlanguage{english}{[10]}
עלול להיות מוצג כ-\foreignlanguage{english}{[01]}
בהיעדר עטיפה מתאימה,
ולכן יש להקיף כל פקודת ציטוט
ב-\foreignlanguage{english}{\texttt{\textbackslash mbox\{\ldots\}}}
בתוך כל סביבה עברית~\mbox{\cite{Segal2026}}.
\end{hebrew}

\section[\texthebrew{פתרון \foreignlanguage{english}{LuaLaTeX} ו-\foreignlanguage{english}{Polyglossia}}]%
        {\texthebrew{פתרון \foreignlanguage{english}{LuaLaTeX} ו-\foreignlanguage{english}{Polyglossia}}\mbox{\setLTR\enspace\thesection}}
\label{sec:bidi-solution}

\begin{hebrew}
חבילת \foreignlanguage{english}{Polyglossia}
בשילוב מנוע \foreignlanguage{english}{LuaLaTeX}
מספקת את הפתרון המלא לתמיכה בעברית ב-\foreignlanguage{english}{LaTeX}~\mbox{\cite{Polyglossia2023}}.
ההגדרה בפרמבל של המסמך כוללת שני צעדים:
\foreignlanguage{english}{\texttt{\textbackslash setmainlanguage\{hebrew\}}}
להכרזת עברית כשפה הראשית,
ו-\foreignlanguage{english}{\texttt{\textbackslash setotherlanguage\{english\}}}
לרישום אנגלית כשפה משנית.
לאחר ההגדרה, \foreignlanguage{english}{Polyglossia}
קובע אוטומטית את כיוון הפסקה כ-\foreignlanguage{english}{RTL}
ומאפשר שימוש בסביבה
\foreignlanguage{english}{\texttt{\textbackslash begin\{hebrew\}\ldots\textbackslash end\{hebrew\}}}
לבלוקים שלמים של טקסט עברי,
ובפקודה
\foreignlanguage{english}{\texttt{\textbackslash texthebrew\{\ldots\}}}
לטקסט עברי מוטמע בתוך פסקה אנגלית~\mbox{\cite{Polyglossia2023}}.

כדי להטמיע מונח אנגלי בתוך בלוק עברי
תוך שמירת כיוון הקריאה,
יש להשתמש בפקודה
\foreignlanguage{english}{\texttt{\textbackslash foreignlanguage\{english\}\{\ldots\}}}.
לדוגמה, כדי לכתוב את שם המסגרת
\foreignlanguage{english}{CrewAI}
בתוך פסקה עברית מבלי שיתהפך,
יש לנסח:
\foreignlanguage{english}{\texttt{\textbackslash foreignlanguage\{english\}\{CrewAI\}}}.
מנוע \foreignlanguage{english}{LuaLaTeX}
תומך בגופנים \foreignlanguage{english}{OpenType}
עם יכולות \foreignlanguage{english}{RTL} מלאות,
ובכך מבטיח עיצוב מקצועי של עמודים עבריים
גם במסמכים טכניים הכוללים קוד ונוסחאות~\mbox{\cite{Segal2026}}.
\end{hebrew}

\section[\texthebrew{אתגרי ייצור בצינור הסוכנים}]%
        {\texthebrew{אתגרי ייצור בצינור הסוכנים}\mbox{\setLTR\enspace\thesection}}
\label{sec:bidi-production}

\begin{hebrew}
בצינור הייצור שתואר בפרק הקודם,
סוכן ה-\foreignlanguage{english}{Writer}
מייצר קוד \foreignlanguage{english}{LaTeX}
המכיל גם טקסט עברי וגם פקודות אנגליות.
הסוכן חייב לקבל הנחיות מפורשות
לעטוף כל מונח טכני בפקודת
\foreignlanguage{english}{\texttt{\textbackslash foreignlanguage\{english\}\{\ldots\}}},
ולא להשאיר תוויות אנגליות חשופות
בתוך
\foreignlanguage{english}{\texttt{\textbackslash begin\{hebrew\}\ldots\textbackslash end\{hebrew\}}}~\mbox{\cite{Segal2026}}.
אחרת,
מנוע ה-\foreignlanguage{english}{UBA}
יהפוך כל רצף \foreignlanguage{english}{ASCII}
חשוף בכיוונו,
מה שמייצר פלט בלתי-קריא ב-\foreignlanguage{english}{PDF}.

אתגר נוסף הוא טיפול בטבלאות עם עמודות מעורבות.
כאשר כותרות הטבלה בעברית
ותוכן תאים כולל שמות פקודות אנגליות,
חבילת \foreignlanguage{english}{Polyglossia}
דורשת שכל תא עם טקסט אנגלי יהיה מוצהר מפורשות
עם
\foreignlanguage{english}{\texttt{\textbackslash foreignlanguage\{english\}\{\ldots\}}};
אחרת, מנוע \foreignlanguage{english}{LuaLaTeX}
יחיל את כיוון ה-\foreignlanguage{english}{RTL}
גם על תאי הטבלה האנגליים~\mbox{\cite{Polyglossia2023}}.

הכלי \foreignlanguage{english}{CrewAI}
מאפשר הגדרת תבנית פלט מובנית
(\foreignlanguage{english}{Pydantic output schema})
לכל סוכן~\mbox{\cite{CrewAI2024}}.
בכך, סוכן ה-\foreignlanguage{english}{Writer}
מייצר תמיד \foreignlanguage{english}{LaTeX}
עם כל הסימונים הנדרשים לדו-כיוניות תקינה.
בדיקת האיכות הופכת לאוטומטית:
אם פלט הסוכן אינו עומד בסכמת ה-\foreignlanguage{english}{Pydantic},
הצינור נעצר לפני הידור ה-\foreignlanguage{english}{PDF}
ומאפשר תיקון מיידי~\mbox{\cite{Anthropic2024b}}.
\end{hebrew}

\section{Mixed-Language Example}
\label{sec:bidi-example}

The following table summarises the primary \LaTeX{} commands used throughout this chapter
to achieve correct bidirectional text rendering with
\texttt{Polyglossia} and \texttt{LuaLaTeX}.
Each command is shown using
\texttt{\textbackslash texttt\{\textbackslash textbackslash cmd\}} notation
in accordance with the document output rules
(no \texttt{\textbackslash verb} is used anywhere in this document).

\begin{center}
\begin{longtable}{@{}p{3.8cm}p{5.8cm}p{4.0cm}@{}}
\toprule
\textbf{Command} &
\textbf{Purpose} &
\textbf{Usage Context} \\
\midrule
\endfirsthead
\toprule
\textbf{Command} &
\textbf{Purpose} &
\textbf{Usage Context} \\
\midrule
\endhead
\texttt{\textbackslash begin\{hebrew\}\ldots\textbackslash end\{hebrew\}} &
  Sets paragraph direction to RTL and activates Hebrew hyphenation and
  \texttt{OpenType} font features \cite{Polyglossia2023}. &
  Full paragraphs of Hebrew prose inside an otherwise LTR document. \\[4pt]
\texttt{\textbackslash texthebrew\{\ldots\}} &
  Inline RTL span within an LTR paragraph; switches language locally without
  altering the surrounding paragraph direction. &
  Short Hebrew phrases or section headings embedded in English text. \\[4pt]
\texttt{\textbackslash foreignlanguage\{english\}\{\ldots\}} &
  Forces LTR rendering for an English word or technical term inside a Hebrew
  block, preventing character reversal by the UBA algorithm
  \cite{Unicode2023}. &
  Technical terms, command names, and proper nouns inside
  \texttt{\textbackslash begin\{hebrew\}\ldots\textbackslash end\{hebrew\}}. \\[4pt]
\texttt{\textbackslash setmainlanguage\{hebrew\}} &
  Declares Hebrew as the primary document language and sets the default paragraph
  direction to RTL \cite{Polyglossia2023}. &
  Document preamble, immediately after
  \texttt{\textbackslash usepackage\{polyglossia\}}. \\[4pt]
\texttt{\textbackslash mbox\{\textbackslash cite\{key\}\}} &
  Prevents the citation bracket and number from being split or reversed by the
  UBA algorithm; critical for multi-digit references such as
  \texttt{[10]} \cite{Unicode2023}. &
  All \texttt{\textbackslash cite\{\}} calls inside Hebrew environments. \\[4pt]
\bottomrule
\end{longtable}
\end{center}
